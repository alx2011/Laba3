CREATE TABLE  "COMPANY" 
   (	"COMP_ID" NUMBER NOT NULL ENABLE, 
	"NAME" VARCHAR2(60) NOT NULL ENABLE, 
	"DESCRIPTION" VARCHAR2(255), 
	 CONSTRAINT "MCOMPANY_PK" PRIMARY KEY ("COMP_ID") ENABLE, 
	 CONSTRAINT "MCOMPANY_CON" UNIQUE ("NAME") ENABLE
   )

CREATE TABLE  "TYPE" 
   (	"TYPE_ID" NUMBER NOT NULL ENABLE, 
	"GAMETYPE" VARCHAR2(20) NOT NULL ENABLE, 
	 CONSTRAINT "GTYPE_PK" PRIMARY KEY ("TYPE_ID") ENABLE, 
	 CONSTRAINT "GTYPE_CON" UNIQUE ("GAMETYPE") ENABLE
   )

CREATE TABLE  "PRODUCT" 
   (	"PRODUCT_ID" NUMBER NOT NULL ENABLE, 
	"COMP_ID" NUMBER NOT NULL ENABLE, 
	"TYPE_ID" NUMBER NOT NULL ENABLE, 
	"NAME" VARCHAR2(60) NOT NULL ENABLE, 
	"DESCRIPTION" VARCHAR2(255), 
	 CONSTRAINT "PRODUCT_PK" PRIMARY KEY ("PRODUCT_ID") ENABLE, 
	 CONSTRAINT "COMP_ID_FK" FOREIGN KEY ("COMP_ID")
	  REFERENCES  "COMPANY" ("COMP_ID") ON DELETE CASCADE ENABLE, 
	 CONSTRAINT "TYPE_ID_FK" FOREIGN KEY ("TYPE_ID")
	  REFERENCES  "TYPE" ("TYPE_ID") ON DELETE CASCADE ENABLE
   )
